with Ada.Text_IO;             use Ada.Text_IO;
with Ada.Characters.Handling; use Ada.Characters.Handling;
with Ada.Characters.Latin_1;  use Ada.Characters.Latin_1;
procedure Main is
   Input : String (1 .. 5);
   Last  : Natural;

   procedure Print (Char : Character) is
   begin
      Put ("Result: ");
      if Is_Letter (Char) then
         if Is_Upper (Char) then
            Put (To_Lower (Char));
         else
            Put (To_Upper (Char));
         end if;
      elsif Is_Graphic (Char) then
         Put (Char);
      elsif Is_Line_Terminator (Char) then
         case Char is
            when LF  => Put ("LF");
            when VT  => Put ("VT");
            when FF  => Put ("FF");
            when CR  => Put ("CR");
            when NEL => Put ("NEL");
            when others =>
               Put ("Missed one: " & Integer'image (Character'pos (Char)));
         end case;
      else
         Put ("unprintable");
      end if;
      New_Line;
   end Print;

begin
   Put ("Enter ASCII value: ");
   Get_Line (Input, Last);
   Print (Character'val (Integer'value (Input (1 .. Last))));
end Main;
